\section{Behaviours}
\subsection{Syntax}
\begin{figure}[h]
\begin{verbatim}
defbehaviour M do
  [ $param X
  | $opaque X X1 ... Xn
  | $type X X1 ... Xn = t
  | callback X : t
  ]*
end
\end{verbatim}
\caption{Surface syntax proposal for behaviours}
\end{figure}
\begin{figure}[h]
  \[D::= \text{param} X | \text{type} X \overline{X} = T | \text{opaque} X

  \caption{Formal syntax}
\end{figure}
